const express = require('express');
const router = express.Router();
const { getSalesDocuments } = require('../services/clearbooksAPI');


// route: GET /clearbooks/sales/:salesType
router.get('/sales/:salesType', async (req, res) => {
  const { salesType } = req.params;
  try {
    const data = await getSalesDocuments(salesType);
    res.json({ success: true, data });
  } catch (err) {
    res.status(500).json({ success: false, message: err.message });
  }
});

module.exports = router;







const axios = require('axios');
require('dotenv').config();

const clearbooksAPI = axios.create({
  baseURL: 'https://api.clearbooks.co.uk/v1',
  headers: {
    'Authorization': `Bearer ${process.env.CLEARBOOKS_ACCESS_TOKEN}`,
    'Content-Type': 'application/json'
  }
});

// ✅ Function to get sales documents
async function getSalesDocuments(salesType) {
  try {
    const response = await clearbooksAPI.get(`/accounting/sales/${salesType}`);
    return response.data;
  } catch (error) {
    console.error('Error fetching sales documents:', error.response?.data || error.message);
    throw error;
  }
}

module.exports = { getSalesDocuments };
